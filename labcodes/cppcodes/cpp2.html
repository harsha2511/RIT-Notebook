Write a C++ program to find the simple interest

#include < iostream >
using namespace std;
int main()
{
float p, r, t, si;
cout<<"Enter Principle Amount: ";
cin>>p;
cout<<"Enter Rate of Interest: ";
cin>>r;
cout<<"Enter Time Period: ";
cin>>t;
si = (p*r*t)/100;
cout<<"\nSimple Interest Amount: "<<si;
cout<<endl;
return 0;
}
