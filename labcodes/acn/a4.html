#server.py
import socket
from ch import ip_checksum

s=socket.socket()
s.bind(('',9999))
s.listen(1)
x,_=s.accept()
m=x.recv(1024)
msg,cs=m.rsplit(b'|',1)
print("Received:",msg)
print("Received checksum:",cs.decode())
print("Verification:",'OK' if hex(ip_checksum(msg))==cs.decode() else 'Error')
x.close()
s.close()

#client.py
import socket
from ch import ip_checksum

s=socket.socket()
s.connect(('localhost',9999))
m=b"Hello"
cs=hex(ip_checksum(m))
s.send(m+b'|'+cs.encode())
s.close()
