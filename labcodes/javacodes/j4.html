Bitwise Operators

class onec {
public static void main(String[] args) {
int number1 = 12, number2 = 25, result;
// bitwise OR between 12 and 25
result = number1 | number2;
System.out.println(result);
// prints 29
// bitwise AND between 12 and 25
result = number1 & number2;
System.out.println(result);
// prints 8
}
}
