class Account{
String accountNumber, Name;
double initial_balance = 12000, deposit, withdraw;
void checkbalance(){
System.out.println("Balance="+initial_balance);
}
void deposition(){
deposit = 5000;
System.out.println("Deposit="+deposit);
System.out.println("Final balance = "+(deposit+initial_balance));
}
void withdrawl(){
withdraw = 6000;
if(initial_balance-withdraw < 3000){
System.out.println("Insufficient balance");
}
else{
System.out.println("Withdrawl = "+withdraw);
System.out.println("Final balance="+(initial_balance-withdraw));
}
}
void display(){
System.out.println("Name = "+Name);
System.out.println("Account No = "+accountNumber);
}
}
