Staff Class

import java.util.Scanner;
class Staff{
int staffId, Phone;
String Name;
float Salary;
}
class Teaching extends Staff{
String domain;
int publications;
void dispa(){
System.out.printf("%d\t%d\t%s\t%f\t%s\t%d\t", staffId, Phone, Name, Salary, domain, publications);
}
}
class Technical extends Staff{
int skills;
void dispb(){
System.out.printf("%d\t", skills);
}
}
class Contract extends Staff{
int period;
void dispc(){
System.out.printf("%d\n", period);
}
}
