rng(1); % Set random seed for reproducibility
n = 100; % Number of data points
t = (1:n)'; % Time points
price_mean = 1000; % Mean stock price
price_std = 5; % Standard deviation
stock_prices = price_mean + price_std * randn(n, 1)
% Create a time series object
ts = timeseries(stock_prices, t, 'Name', 'Stock Prices');
% Plot the historical stock prices
figure;
plot(ts.Time, ts.Data);
title('Historical Stock Prices');
xlabel('Time');
ylabel('Stock Price');
% Perform time series analysis and prediction (simple moving average)
window_size = 5; % Moving average window size
moving_avg = movmean(ts.Data, window_size); % Simple moving average
% Plot the original stock prices and moving average
figure;
plot(ts.Time, ts.Data, 'b', 'DisplayName', 'Stock Prices');
hold on;
plot(ts.Time, moving_avg, 'r', 'DisplayName', 'Moving Average');
title('Stock Prices vs. Moving Average');
xlabel('Time');
ylabel('Value');
legend('Location', 'northwest');
% Predict future stock prices (simple extrapolation)
forecast_horizon = 10; % Number of days to forecast
last_price = ts.Data(end); % Last observed stock price
meanvalue = mean(diff(ts.Data))
forecasted_prices = last_price + (1:forecast_horizon) * meanvalue
% Plot the original stock prices and forecasted prices
figure;
plot(ts.Time, ts.Data, 'b', 'DisplayName', 'Stock Prices');
hold on;
plot(ts.Time(end) + (1:forecast_horizon), forecasted_prices, 'r', 'DisplayName', 'Forecasted Prices');
title('Stock Prices vs. Forecasted Prices');
xlabel('Time');
ylabel('Value');
legend('Location', 'northwest');
