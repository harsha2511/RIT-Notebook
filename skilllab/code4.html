% Sample diabetes dataset (replace this with your actual dataset)
% Define the file path to your dataset
file_path = "D:\ML2023\Matlab2023\Diabetics1.csv";
% Load the dataset from the CSV file
data = readtable(file_path);
% Extract the blood pressure data
bloodPressure = data{:, 4};
% Create a histogram with custom settings
histogram(bloodPressure, 'BinWidth', 10, 'FaceColor', 'blue', 'EdgeColor', 'black');
% Add labels and title
xlabel('Blood Pressure');
ylabel('Frequency');
title('Blood Pressure Histogram in Diabetic Dataset');
