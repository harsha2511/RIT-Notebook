age_groups = {'Young', 'Middle-Aged', 'Elderly'};
age_thresholds = [0, 35, 60, 90]; % Define age thresholds for age groups

% Categorize individuals into age groups
data.AgeGroup = categorical(discretize(data.Age, age_thresholds, 'categorical', age_groups));
% Use the 'convertvars' function to replace the original 'Age' variable with the 'AgeGroup' variable
data = convertvars(data, 'AgeGroup', 'categorical')
% Summary statistics by age group
summary_stats = groupsummary(data, 'AgeGroup', 'mean', ['Outcome','BloodPressure',"Glucose"]);
% Display the summary statistics
disp(summary_stats);
% Create bar plots to visualize diabetes status by age group
figure;
bar(summary_stats.AgeGroup, summary_stats.mean_Outcome);
xlabel('Age Group');
ylabel('Mean Diabetes Status (1=Diabetic, 0=Non-Diabetic)');
title('Diabetes Status by Age Group');
% Create bar plots to visualize mean Glucose levels by age group
figure;
bar(summary_stats.AgeGroup, summary_stats.mean_Glucose);
xlabel('Age Group');
ylabel('Mean Glucose Level');
title('Mean Glucose Levels by Age Group');
% Create bar plots to visualize mean Blood Pressure by age group
figure;
bar(summary_stats.AgeGroup, summary_stats.mean_BloodPressure);
xlabel('Age Group');
ylabel('Mean Blood Pressure');
title('Mean Blood Pressure by Age Group');
