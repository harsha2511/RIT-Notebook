rng('default'); % For reproducibility
transactionAmounts = randn(1000, 1) * 100 + 500 % Example data
figure;
% Create a histogram with custom settings
histogram(transactionAmounts, 'BinWidth', 50, 'FaceColor', 'blue', 'EdgeColor', 'black');
% Add labels and title
xlabel('Transaction Amount ($)');
ylabel('Frequency');
title('Credit Card Usage Histogram');
% Optionally, display count above each bar
h = histogram(transactionAmounts, 'BinWidth', 50);
counts = h.Values;
text(h.BinEdges(1:end-1) + h.BinWidth/2, counts, arrayfun(@(x) num2str(x), counts, 'UniformOutput', false), 'HorizontalAlignment', 'center', 'VerticalAlignment', 'bottom');
