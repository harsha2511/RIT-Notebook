file_path = "D:\ML2023\Matlab2023\Diabetics1.csv";
% Load the dataset from the CSV file
data = readtable(file_path);
is_diabetic = data{:, 9};
Blood_PressureValues = data{:,4}
% Separate the Blood_Pressure for diabetic and non-diabetic groups
bmiDiabetic = Blood_PressureValues(is_diabetic == 1)
bmiNonDiabetic = Blood_PressureValues(is_diabetic == 0)
% Create a histogram to visualize the BMI distribution
figure;
histogram(bmiDiabetic, 'BinWidth', 2, 'FaceColor', 'blue', 'EdgeColor', 'black');
title('BMI Distribution in Diabetic Patients');
xlabel('BMI');
ylabel('Frequency');
figure;
% Create a histogram to visualize the BMI distribution
figure;
histogram(bmiNonDiabetic, 'BinWidth', 2, 'FaceColor', 'blue', 'EdgeColor', 'black');
title('BMI Distribution in Diabetic Patients');
xlabel('BMI');
ylabel('Frequency');
