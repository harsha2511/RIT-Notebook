bookTitles = {'Book 1', 'Book 2', 'Book 3', 'Book 4', 'Book 5'};
authors = {'Author A', 'Author B', 'Author C', 'Author A', 'Author D'};
publicationDates = {'2020-01-01', '2019-03-15', '2018-05-20', '2020-01-01', '2017-10-05'};
availableCopies = [300, 288, 58888, 1888, 4666];
libraryCatalog = table(bookTitles', authors', publicationDates', availableCopies', ...
'VariableNames', {'Title', 'Author', 'PublicationDate', 'AvailableCopies'});
% Display the library catalog
disp(libraryCatalog);
% Search function to find books by title or author
searchTitle = 'Book 1'; % Search for a book by title
searchAuthor = 'Author A'; % Search for books by author
matchingBooksTitle = libraryCatalog(strcmpi(libraryCatalog.Title, searchTitle), :);
matchingBooksAuthor = libraryCatalog(strcmpi(libraryCatalog.Author, searchAuthor), :);
% Display search results
fprintf('Books matching title "%s":\n', searchTitle);
disp(matchingBooksTitle);
fprintf('Books by author "%s":\n', searchAuthor);
disp(matchingBooksAuthor);
% Identify popular books (available copies greater than a threshold)
popularThreshold = 3;
popularBooks = libraryCatalog(libraryCatalog.AvailableCopies > popularThreshold, :);
fprintf('Popular books (Available Copies > %d):\n', popularThreshold);
disp(popularBooks);
figure;
% Create a bar plot to show the number of available copies for each book
bar(libraryCatalog.Title, libraryCatalog.AvailableCopies);
title('Available Copies by Book');
xlabel('Book Titles');
ylabel('Available Copies');
figure;
% Filter books with fewer than 2 available copies
fewCopiesBooks = libraryCatalog(libraryCatalog.AvailableCopies < 2000, :);

% Create a bar plot to display the number of available copies for these books
bar(fewCopiesBooks.Title, fewCopiesBooks.AvailableCopies);
title('Books with Fewer than 2 Available Copies');
xlabel('Book Titles');
ylabel('Available Copies');
% Rotate the x-axis labels for better readability
xtickangle(45);
