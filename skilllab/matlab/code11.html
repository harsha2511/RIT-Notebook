numDays = 30; % Number of days
rng('default'); % Initialize random number generator for reproducibility
% Generate random data for temperature, humidity, and precipitation
temperature = randi([50, 100], numDays, 1); % Random temperature between 50 and 100 (Fahrenheit)
humidity = randi([20, 80], numDays, 1); % Random humidity between 20% and 80%
precipitation = randi([0, 10], numDays, 1); % Random precipitation in inches
% Create a table to store daily weather data
weatherTable = table(temperature, humidity, precipitation, ...
'VariableNames', {'Temperature', 'Humidity', 'Precipitation'});
% Calculate the average temperature
averageTemperature = mean(weatherTable.Temperature);
% Identify days with heavy rainfall (precipitation greater than a threshold)
heavyRainThreshold = 4; % Set the threshold for heavy rainfall
heavyRainDays = find(weatherTable.Precipitation > heavyRainThreshold);
% Display the weather data table, average temperature, and heavy rain days
disp(weatherTable);
fprintf('Average Temperature: %.2fÂ°F\n', averageTemperature);
fprintf('Days with Heavy Rain (Precipitation > %.2f inches):\n', heavyRainThreshold);
disp(heavyRainDays);
